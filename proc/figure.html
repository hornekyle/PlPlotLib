	subroutine figure
		!! Create a new figure
		logical,save::isFirst = .true.
		
		if(.not.isFirst) then
			call pleop()
		else
			isFirst = .false.
		end if
		
		call plbop()
		call plssub(1,1)
		call pladv(1)
		call resetPen()
	end subroutine figure
